## 📋 PART 5: IMPLEMENTATION GUIDE

### **Three.js Soul Character Model**

```javascript
// Basic soul character structure
class SoulCharacter3D {
    constructor(type, tier) {
        this.type = type; // 'profit', 'love', 'tax', 'hybrid'
        this.tier = tier; // 'common', 'uncommon', 'rare', 'epic', 'legendary'
        
        // Shape language
        this.shape = this.getShapeForType(type);
        
        // Proportions
        this.headsTall = this.getHeadsTall(tier);
        
        // Colors
        this.colors = this.getColorsForType(type, tier);
        
        // Create 3D model
        this.createModel();
    }
    
    getShapeForType(type) {
        const shapes = {
            'profit': 'triangle',
            'love': 'circle',
            'tax': 'square',
            'hybrid': 'star'
        };
        return shapes[type];
    }
    
    getHeadsTall(tier) {
        const tiers = {
            'common': 4,
            'uncommon': 4.5,
            'rare': 5,
            'epic': 5.5,
            'legendary': 6
        };
        return tiers[tier];
    }
    
    getColorsForType(type, tier) {
        const palettes = {
            'profit': ['#FFD700', '#FFAA44', '#8B4513'],
            'love': ['#FFB6C1', '#FF69B4', '#FF1493'],
            'tax': ['#88AADD', '#4682B4', '#708090'],
            'hybrid': ['#DDDDDD', '#8888FF', '#FF88FF']
        };
        
        // Legendary souls get gold accent
        if (tier === 'legendary') {
            palettes[type].push('#FFD700');
        }
        
        return palettes[type];
    }
    
    createModel() {
        // Create Three.js geometry based on shape
        const geometry = this.shape === 'circle' 
            ? new THREE.SphereGeometry(1, 32, 32)
            : this.shape === 'square'
            ? new THREE.BoxGeometry(1.5, 1.5, 1.5)
            : new THREE.ConeGeometry(1, 2, 4); // Triangle
        
        const material = new THREE.MeshStandardMaterial({
            color: this.colors[0],
            emissive: this.colors[1],
            emissiveIntensity: 0.3
        });
        
        this.mesh = new THREE.Mesh(geometry, material);
    }
}
```
